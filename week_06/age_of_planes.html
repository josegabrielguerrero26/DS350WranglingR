## Join by tail number
dfcom <- flights%>%
        left_join(planes,c("tailnum"="tailnum"))

#calculate the age of the plane that was used.
dfcom<-mutate(dfcom, age=year.x - year.y)
dfcom <-dfcom%>% 
  rename(creation_year = year.y, dept_year = year.x)

## Chart
Create a chart that looks at the distribution of departure delays by age (or age group). Do you notice any patterns?
Create a new column that groups age into 0-10, 11-20, 21-30, and >30

```{r}
#| label: chart1
#| echo: false
chart_1<- mutate(dfcom, age_references=case_when(
    age>= 0 & age <= 10 ~ "0-10",
    age>= 11 &age <= 20 ~ "11-20",
    age>= 21 & age <= 30 ~ "21-30",
    age> 30 ~ ">30",
    TRUE ~ "others"  
))

table <- select(chart_1, age,age_references,creation_year, dept_year, dep_delay)



sum <- table%>%
  group_by(age_references)%>%
 summarise(tot_delays = sum(dep_delay > 0, na.rm = TRUE)) 
    
ggplot(sum, aes(x = age_references, y = tot_delays, fill = age_references)) +
  theme_light()+
  geom_bar(stat = "identity")+
  labs(title = "Flights", y = "Delay Time", x = "Group Ages")+
  theme(plot.title = element_text(hjust = 0.5),
        legend.title = element_blank(),
        panel.grid.major.x=element_blank(),
        panel.grid.minor.y = element_blank()
        )+
scale_y_continuous(breaks =seq(0,50000, by=10000))

#head(table, 5)
```

## Create a chart that looks at the distribution of age by carrier.

```{r}
#| label: bycarrier
#| echo: false

join_carrier<- flights%>%
  left_join(planes, by = "tailnum") %>%
  left_join(airlines, by = "carrier") %>%
  mutate(age=year.x - year.y)

ggplot(join_carrier, aes(x=name, y=age, fill=name))+
geom_boxplot()+
theme_classic()+
labs(title = "Carrier Companies and Age of Planes", y = "Age Planes", x = "Carrier")+
  theme( axis.text.x = element_blank(),
         axis.ticks.x =element_blank(),
  )+
  scale_y_continuous(breaks =seq(0,40, by=10))

```
I see that most of the planes are fairly new between 1 to 15 years. It is interesting that among all carrier there is not a standard for replacing airplanes, which I see this in the distribution of the boxplot.
