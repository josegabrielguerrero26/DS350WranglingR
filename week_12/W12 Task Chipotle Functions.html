```{r}
test <- '{"Monday":94,"Tuesday":76,"Wednesday":89,"Thursday":106,"Friday":130,"Saturday":128,"Sunday":58}'

# function
visits <- function(string) {
  matches <- gregexpr('"(\\w+)":(\\d+)', string)
  matches <- regmatches(string, matches)[[1]]
  
#table 
  result <- matrix(nrow = length(matches), ncol = 2)
  colnames(result) <- c("day", "visits")
  
  for (i in seq_along(matches)) {
    result[i, ] <- strsplit(matches[i], ":")[[1]]
  }
  
  result <- as.data.frame(result)
  result$day <- gsub('"', '', result$day)
  result$visits <- as.integer(result$visits)
  
  return(result)
}

result <- visits(test)
result

```
