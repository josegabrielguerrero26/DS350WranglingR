```{r}
#| echo: false

mpd <- function(string) {
  result <- visits(string)
  if (nrow(result) == 0) {
    return(NULL)
  }
  max_vis <- max(result$visits, na.rm = TRUE)
  pd <- result[result$visits == max_vis, "day"] %>%
    unlist()
  return(paste(pd, collapse = ", "))
}

df2 <- dfp %>%
  mutate(df2 = map_chr(popularity_by_day, mpd))


result <- df2 %>%
  separate_rows(df2, sep = ',\\s*') %>%
  group_by(day = factor(trimws(df2))) %>%
  summarize(count = n(), .groups = 'drop') %>%
  mutate(day = factor(day, levels = c('Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday', 'Sunday')))

result


result %>%
  ggplot(aes(x = day, y = count)) +
  geom_col(color="grey") +
  theme_classic()+
  labs(title = "Days - Counts", x = "", y = "Counts") +
  theme_bw()

```
